```java
class Solution {
    public int climbStairs(int n) {
        if (n == 1) {
            return 1;
        }
        // 设f[i]为到达第i级楼梯的方式数，要到达第i级楼梯由两种方式：
        // 一种是在第i - 2级跨两步，另一种是在第i - 1级跨一步。
        // f[i] = f[i - 2] + f[i - 1]
        // f[0] = 0, f[1] = 1, f[2] = 2
        int[] f = new int[n + 1];
        f[0] = 0;
        f[1] = 1;
        f[2] = 2;
        for (int i = 3; i <= n; i++) {
            f[i] = f[i - 2] + f[i - 1];
        }
        return f[n];
    }
}
```
